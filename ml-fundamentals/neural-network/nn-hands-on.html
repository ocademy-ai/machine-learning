14.2.1. Neural network with Tensorflow : Face recognition#

import numpy as np
import matplotlib.pyplot as plt
import pandas as pd

# Load data
from sklearn.datasets import fetch_lfw_people
faces = fetch_lfw_people(min_faces_per_person=200, resize=0.25)

# 766 images of 31 * 23 pixel black & white
print(faces.images.shape)

(766, 31, 23)

# 2 different target classes
np.unique(faces.target)

array([0, 1])

14.2.1.1. Minimal preprocessing#

# Flatten our 766 images
X = faces.images.reshape(766, 31*23)
X.shape

(766, 713)

y = faces.target
y.shape

(766,)

# Train test split
from sklearn.model_selection import train_test_split
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.25, random_state=3)

# Standardize
from sklearn.preprocessing import StandardScaler
scaler = StandardScaler()
X_train = scaler.fit_transform(X_train)
